MAIL-START()

Description
Returns the value TRUE (non-zero) if a new mail message is successfully started otherwise FALSE (zero)
Syntax
MAIL-START(text)
Category
Environment
Runtime version
4.1 and above
procedure main
   field ws-mail-open type boolean // Sending an email with an attachment
   set ws-mail-open = mail-start("TEST EMAIL")
   set ws-mail-open = mail-add-line(concat("Hi ",str-concat(login-id),","))
   set ws-mail-open = mail-add-line(" ")
   set ws-mail-open = mail-add-line("Please find enclosed the changes to the draft plans we discussed last week.")
   set ws-mail-open = mail-add-line("Note that we allowed an extra week in the schedule because we know you will ")
   set ws-mail-open = mail-add-line("change your mind again." )
   set ws-mail-open = mail-add-line(" ")
   set ws-mail-open = mail-add-line("Kind Regards, The Architect.")
 // Attachment
 // This assumes Pronto Xi is running locally on a Microsoft Windows platform, 
 // and for this case, the in-body flag must be zero. The attachment file
 // c:\tmp\plans.xls must exist, otherwise the message is not sent. 
 // The description of the attachment is visible in the email header, 
 // and should include the file extension so the receiving PC will 
 // know how to open it.
   if mail-attach("c:\tmp\plans.xls" ,"Draft Plans for Renovations.xls",0,0,0)
   endif
   if not mail-send(login-id(),FALSE,FALSE)
   message "This email was not sent - Please ensure the attachment exists"
   endif
   set ws-mail-open = not mail-cancel()
end-procedure
// start the mail message.
if mail-start(lf-screen-name)
endif
// Format the mail message
report lf-screen-name form 'PRONTO_MAIL' full-xml no-message spool-only
report section is 'mail_body' print str-concat(lf-screen-name, translate " - Outstanding Invoices: ")
skip
print concat(translate "Attached file :", lp-file-name)
print concat(translate "Account :", lp-auth-accountcode) set lf-spoolfile = spool-file-name()
report section finished
report finished
if mail-attach(lf-spoolfile,'The mail message',1,1,1) // where the flags 1,1,1 mean // 1 attached file is to be in the body of the mail // 1 convert to HTML // 1 remove the file after it is attached.
endif if mail-attach(lp-file-name,'The mail message',0,0,0) // where the flags 0,0,0 mean // 0 attached file is to be a separate attachment in the mail // 0 do not convert to HTML // 0 do not remove the file after it is attached.
endif if mail-send(lp-email-address,TRUE,FALSE) // where the flag TRUE, FALSE mean // TRUE Flag the mail as cancelled, for example, cannot send // mail to another user. That is, no need to do a mail-cancel() // FALSE Do not send as registered mail
endif
