OLE code examples

This section contains examples of code that use the OLE related function calls provided in the Pronto RAD 4GL language:

For further examples, see the MSDN website at https://msdn.microsoft.com/en-au/default.aspx.

Call an Object Interface

This code example calls an object interface:

///////////////////////////////////////////////////
//
// Sample 4GL program for OLE automation
//
///////////////////////////////////////////////////
#define REFID "{FF045A0F-0546-11D4-B9A7-00104B48BC81}"
        
#define APPNAME "Comsrv.comsrvcls.1"
#define CLSCTX_INPROC_SERVER 1
#define CLSCTX_LOCAL_SERVER 4
        
field inst-id       type ole-interface 
field advise-cookie type ole-interface
field event-cookie  type ole-interface
field dummy-ret     pic 999
field ret           pic s9(8)
field i             pic 99

procedure main
        
  set inst-id = ole-get-active-object(APPNAME)
  if inst-id <= 0
    set inst-id = ole-create-instance(APPNAME, CLSCTX_INPROC_SERVER)
    if inst-id <= 0
      abort "Cannot create instance - error ", ole-status()
    end
  end

  set advise-cookie = ole-advise-event(inst-id, REFID, 32)
  if advise-cookie <= 0
    abort "advise-event failed - ",
          ole-error-description(ole-status())
  end

  // Call the COMM server
  if not ole-call-method(advise-cookie, "test1", dummy-ret)
    message "call-method test1 failed - "
          ole-error-description(ole-status())
  end
  for i = 0 to 19
    if not ole-call-method(advise-cookie, "test2", dummy-ret, i)
        message "call-method test2 failed - "
            ole-error-description(ole-status())
     end
  end

  // Check to see what events we got back
  message "Press enter for events"
  repeat
    set event-cookie = ole-get-event(inst-id, advise-cookie)
  until event-cookie <= 0
    if not ole-get-property(event-cookie, "DispID", ret)
    message "Could not get DispID - ",
            ole-error-description(ole-status())
    end
    message "Event display ID is ", ret
        
    if not ole-get-property(event-cookie, "InvokeKind", ret)
      message "Could not get InvokeKind - ",
            ole-error-description(ole-status())
    end
    message "Event InvokeKind is ", ret
    if not ole-get-property(event-cookie, "ParamCount", ret)
      message "Could not get ParamCount - ",
            ole-error-description(ole-status())
    end
    message "Event ParamCount is ", ret
    for i = 1 to ret
    // Can we determine ParamType?
      if not ole-get-property(event-cookie, "ParamItem", ret, i)
        message "Could not get ParamItem - ",
              ole-error-description(ole-status())
      end
      message "Event ParamItem is ", ret
    end
    if ole-release(event-cookie)
    end
    message "Event id was ", event-cookie
  end-repeat
  pause
  if ole-unadvise-event(inst-id, advise-cookie)
  endif
  if ole-release(advise-cookie)
  endif
  if ole-release(inst-id)
  endif
end-procedure //main ---------------
Embedding a calendar in a Pronto Xi screen

This code example embeds a calendar inside a Pronto Xi screen. This code example also contains samples of adjusting the calendar properties (for example, the background colour):

/////////////////////////////////////////////////////////
//
//
// Sample 4GL program for OLE automation - Calendar Control
//
//
/////////////////////////////////////////////////////////
        
field id     type ole-interface
      colour pic 9(10)

procedure main
  no-hide
  
  set id = ole-create-control("MSCAL.Calendar.7", 5, 5, 8, 30)
  if id = 0
    message "error ID = 0"
    exit
  endif
  pause
  if not ole-get-property(id, "BackColor", colour)
    abort "Could not get BackColor - "
        ole-error-description(ole-status())
  endif
  set colour = colour / 2
  if not ole-put-property(id, "BackColor", colour)
    abort "Could not put BackColor"
  endif
  pause
end-procedure //main ---------------
Embedding an Internet Explorer window in a Pronto Xi screen

This code example embeds an Internet Explorer (IE) window inside a Pronto Xi screen using a Pronto Xi prompt to enter the required internet address:

/////////////////////////////////////////////////////////
//
//
// Sample 4GL program for OLE automation - Internet Explorer
//
//
/////////////////////////////////////////////////////////

field id       type ole-interface
      ret      pic s9(10)
      web-site pic x(256)

screen main
  allowed entry
  form-entry
before
  set id = ole-create-control("Shell.Explorer", 2, 1, 20, 80)
  if id = 0
    message "Cannot create control - "
          ole-error-description(ole-status())
    exit
  endif
  set web-site = "https://www.pronto.net"
detail
  accept web-site @1,30 pic x(40) title "URL:"
    default is web-site
  confirm auto
  confirmed
    if screen-mode = entry
      if not ole-call-method(id, "Navigate2", ret, web-site)
        message "Navigate2 failed - "
          ole-error-description(ole-status())
      endif
    endif
  end
after
  if ole-release(id)
  endif
end-screen //main ---------------------
Embedding a PDF File inside a Pronto Xi Screen

This code example opens a PDF (.pdf) file in an Internet Explorer (IE) window, which is embedded inside a Pronto Xi screen:

////////////////////////////////////////////////////////
//
//
// Sample 4GL program for OLE automation - PDF File
//
//
////////////////////////////////////////////////////////
        
field id       type ole-interface
field ret      pic s9(10)
field web-site pic x(256)

screen main
  window
    title 'Stock Item Specs'
  allowed SEARCH
  primary stock-master
before
  display 'Item : ' @2,2
  box @4.9,2.8 to @23,79.5 sunken colour black on white
      absolute-coordinates
  set id = ole-create-control("Shell.Explorer", 4, 2, 18, 76.5)
  if id = 0
    message "Cannot create control - "
        ole-error-description(ole-status())
    exit
  endif
  set web-site = 'C:\Documents and Settings\My Documents\sample.pdf'
detail
  accept stock-code @2,12
  display stk-description @2,32
  
  if not ole-call-method(id, "Navigate2", ret, web-site)
      message "Navigate2 failed - " ole-error-description(ole-status())
  endif
  confirm auto
  confirmed
  end-confirm
after
  if ole-release(id)
  endif
end-screen //main ----------------------
Microsoft Excel Code and Functions

This code example contains Microsoft Excel code using code libraries and select functions (from Microsoft Excel) to manipulate Pronto Xi data to/from Microsoft Excel spreadsheets:

/////////////////////////////////////////
// Sample 4GL program for OLE automation - Excel //
// For a Full list of methods and properties available - // See microsoft excel
// [ Open excel > Tools > Macros > VB Editor > Help > VB Help ]
//
////////////////////////////////////////////

#define APPNAME "Excel.Application"

field
  wb-id       type ole-interface
  ws-path     pic x(256)
  ws-alpha    pic x(30)
  ws-numeric  pic s9(8)v9999

//link "../../clib/clib100.spl"
link "clib/clib100.spl"
#include "../../../include/i8excel.spl"

procedure main
  // Start Excel
  do clib100-start-excel
    returning
      i8excel-error
      i8excel-instance-id

  // Reset all ole controls if there is an error 
  I8EXCEL_ERROR_HANDLE(TRUE)

  //Add a workbook - create a new spreadsheet
  do clib100-add-new-workbook
    parameter i8excel-instance-id SPACES
    returning i8excel-error  i8excel-workbook-id
  I8EXCEL_ERROR_HANDLE(TRUE)

  //Open a workbook //do clib100-open-workbook // parameter i8excel-instance-id "C:\SAMPLE.XLS" // returning i8excel-error wb-id //if i8excel-error != space //This should fail unless file exists.. //endif

  //Make the workbook visible
  set i8excel-xl = ole-put-property(i8excel-instance-id, 
                                    "Application.Visible", TRUE)

  //Make sheet current, add it if it does not exist
  do clib100-add-new-worksheet
    parameter i8excel-instance-id 1
    returning i8excel-error i8excel-worksheet-id
  I8EXCEL_ERROR_HANDLE(TRUE)

  // Name the sheet - "Sample"
  set i8excel-xl = ole-put-property(i8excel-worksheet-id,"Name","Sample")

  // Now to work with any data, a cell or range of cells needs to be selected. // Once this is done it is possible to read or write to the cell (or cells) // Some libraries have been made to write to and read from a cell // To clear cells...
  if not ole-call-method(i8excel-worksheet-id, "Cells.Clear", 
                         i8excel-dummy-ret)
  end

  // To populate cells
  do clib100-fill-a-cell parameters i8excel-worksheet-id "A1" "Sample"
  do clib100-fill-a-cell parameters i8excel-worksheet-id "A2" "45"

  // To read cells
  do clib100-get-alpha-cell parameters i8excel-worksheet-id "A1" 
    returning ws-alpha
  do clib100-get-numeric-cell parameters i8excel-worksheet-id "A2" 
    returning ws-numeric 
  message "These values were retrieved from A1 and A2 >" ws-alpha " & 
          " ws-numeric 

  // To format cells, first select the cells or range of cells.
  if ole-get-property(i8excel-worksheet-id, "Range", i8excel-range-id, 
                      "A1", "A2")
  // These statements format the cells
    set i8excel-xl = ole-put-property(i8excel-range-id,
                                     "Font.Bold", 1)
    set i8excel-xl = ole-put-property(i8excel-range-id,
                                      "Font.Size", 20)
    set i8excel-xl = ole-put-property(i8excel-range-id,
                                      "Font.Name","Arial")
    set i8excel-xl = ole-put-property(i8excel-range-id,
                                      "HorizontalAlignment", xlCenter)
    set i8excel-xl = ole-put-property(i8excel-range-id,
                                      "Interior.ColorIndex", 24)

    // A border drawing library has been created as well // (Constants are found in i8excel) 
    do clib100-draw-border
      parameter
        i8excel-range-id 
        xlEdgeTop
        xlContinuous
        1
        xlThick 
    do clib100-draw-border
      parameter
        i8excel-range-id 
        xlEdgeBottom
        xlContinuous
        1
        xlThin 

    // To autofit the cells...
    set i8excel-xl = ole-call-method(i8excel-range-id,
                                     "Columns.AutoFit", i8excel-dummy-ret)

    // Must always remember to release the selected cells
    set i8excel-xl = ole-release(i8excel-range-id)
  endif

  // Other formatting // This will set the top 4 rows as titles on each page.
  set i8excel-xl = ole-put-property(i8excel-worksheet-id,
                                    "PageSetup.PrintTitleRows", "$1:$4")

  // Other selection properties // This will select all cells used.
  if ole-get-property(i8excel-worksheet-id, "UsedRange", i8excel-range-id)
    // Can use this to unprotect protected cells.
    if not ole-call-method(i8excel-worksheet-id,"UnProtect",
                           i8excel-dummy-ret)
    endif

    // Can do a global replace on a sheet.. (A smaller range could be used)
    if not ole-call-method(i8excel-range-id,"Replace", i8excel-dummy-ret,
                           "Sample" "Replaced Sample")
    endif
  set i8excel-xl = ole-call-method(i8excel-range-id, "Columns.AutoFit",
                                     i8excel-dummy-ret)
    set i8excel-xl = ole-release(i8excel-range-id)
  endif

  // Sometimes Excel will pop up alerts; the alerts can be disabled // with this statement
  set i8excel-xl = ole-put-property(i8excel-instance-id,
                                    "Application.DisplayAlerts", FALSE)

  // To minimise the sheet...
  set i8excel-xl = ole-put-property(i8excel-instance-id,
                                    "Application.WindowState",xlMinimized)

  // To maximise the sheet...
  set i8excel-xl = ole-put-property(i8excel-instance-id,
                                    "Application.WindowState",xlMaximized)

  // To save the workbook
  set i8excel-xl = ole-call-method(i8excel-instance-id,
                                   "Application.GetSaveAsFilename",
                                   ws-path, "C:\TryThis.xls")
  set i8excel-xl = ole-call-method(i8excel-workbook-id, "SaveAs",
                                   i8excel-dummy-ret, ws-path) 
  set i8excel-xl = ole-call-method(i8excel-workbook-id, "Save",
                                   i8excel-dummy-ret)

  // Some printing functions. // Set the default printer...
  set i8excel-xl = ole-call-method(i8excel-worksheet-id, "PrintOut",
                                   i8excel-dummy-ret)
  
  set i8excel-xl = ole-put-property(i8excel-instance-id,
                                    "Application.DisplayAlerts", TRUE)

  // To close the workbook
  confirm
     prompt "Do you want to close the workbook"
  confirmed  
     set i8excel-xl = ole-call-method(i8excel-workbook-id, Close",
                                     i8excel-dummy-ret )
  end-confirm

  // Always release ole controls 
  if ole-release(i8excel-worksheet-id)
  endif
  if ole-release(i8excel-workbook-id)
  endif
  if ole-release(i8excel-instance-id)
  endif
  end-procedure //main ---------------------
Inserting a Calendar Event or Task into MS Outlook

This code example inserts a calendar event or task event into MS Outlook:

/////////////////////////////////////////////////////
// Sample 4GL program for OLE automation - Outlook /////////////////////////////////////////////////////
#define APPNAME "Outlook.Application"
        
field outlook-id     type ole-interface
field task-folder-id type ole-interface
field calendar-id    type ole-interface
field namespace-id   type ole-interface
field item-id        type ole-interface
field item-no        type numeric

field ver            pic x(60)
field dummy-ret      pic 999
field ws-a           pic x(40)
field ws-b           type date-time
field ws-c           type date-time
field ws-d           pic x(40)

field ws-range-id    pic s9(8)
field ws-xl          type boolean
field q              pic 9
field ws-path        type string pic x(256)
field i              pic 999

procedure main
  set outlook-id = ole-get-active-object(APPNAME)
  if outlook-id <= 0
    set outlook-id = ole-create-instance(APPNAME, CLSCTX_LOCAL_SERVER)
    if outlook-id <= 0
      abort "Cannot create instance - error ", ole-status()
    end
  end
  if not ole-get-property(outlook-id, "Application.Version", ver)
    abort "Could not get version property: ", ole-status()
  end
  if not ole-get-property(outlook-id, "GetNameSpace",namespace-id,"MAPI")
    abort "No namespace"
  endif
  // // Insert Calendar Item   
  if ole-call-method(outlook-id,"CreateItem",item-id,1)
    set ws-xl = ole-put-property(item-id,"Body","The email text.")
    set ws-xl = ole-put-property(item-id, "Start", concat(str(day(today())),
                                 "/",str(month(today())), "/",
                                 str(year(today())), " 8:00:00 PM"))
    set ws-xl = ole-put-property(item-id, "End", concat(str(day(today())),
                                 "/", str(month(today())), "/",
                                 str(year(today())), " 8:30:00 PM"))
    set ws-xl = ole-put-property(item-id,"ReminderMinutesBeforeStart",30)
    set ws-xl = ole-put-property(item-id,"Subject",
                                 "OLE inserted Calendar Item")
   set ws-xl = ole-call-method(item-id,"Save",dummy-ret)
  endif
  // // Insert Task Item   
  if ole-call-method(outlook-id,"CreateItem",item-id,3)
    set ws-xl = ole-put-property(item-id,"StartDate",today())
    set ws-xl = ole-put-property(item-id,"DueDate",today())
    set ws-xl = ole-put-property(item-id,"Body",
                                 "Ole inserted Task - Body of the task")
    set ws-xl = ole-put-property(item-id,"Sensitivity",0)
    set ws-xl = ole-put-property(item-id,"ReminderSet",-1)
    set ws-xl = ole-put-property(item-id,"ReminderTime",
                                 concat(str(day(today())), "/",
                                 str(month(today())), "/",
                                 str(year(today()))," 8:30:00 AM"))
    set ws-xl = ole-put-property(item-id,"Subject","OLE inserted Task Item")
    set ws-xl = ole-call-method(item-id,"Save",dummy-ret)
  endif
  
  set ws-xl = ole-release(outlook-id)
  set ws-xl = ole-release(item-id)
  set ws-xl = ole-release(namespace-id)
end-procedure //main -------------------------------------
Populating Data and Creating a Chart

This code example is similar to the Populating an Excel Spreadsheet and Creating a Chart example, but uses Microsoft components instead of Excel components:

//////////////////////////////////////////////////
//
// Sample 4GL program for OLE automation - MS Graph
//
//////////////////////////////////////////////////

#define APPNAME "MSGraph.Chart"
#define CLSCTX_LOCAL_SERVER 4

field cookie type ole-interface
field ver    pic x(60)

procedure main
  set cookie = ole-get-active-object(APPNAME)
  if cookie <= 0
    set cookie = ole-create-instance(APPNAME, CLSCTX_LOCAL_SERVER)
    if cookie <= 0
      message "Cannot create instance - error ", ole-status()
      exit
  end
  end
  if not ole-get-property(cookie, "Application.Version", ver)
    message "Could not get version property: ", ole-status(), " - ",
        ole-error-description(ole-status())
    exit
  end
  do fill-sheet-font
  do fill-sheet
  if not ole-put-property(cookie, "Application.Visible", 1)
    message "Could not put-property Visible"
  end
  pause
  if ole-release(cookie)
  endif
  end-procedure //main --------------

procedure fill-sheet-font
local field axes-id type ole-interface

    message "put-property error for Italic"
  endif

    message "put-property error for Size"
  endif
  if not ole-put-property(cookie, "Application.Datasheet.Font.Name", "Tahoma")
    message "put-property error for name"
  endif
  
  if not ole-put-property(cookie, "Application.Chart.HasTitle", TRUE)
    message "put-property error for has chart title text"
  endif
  if not ole-put-property(cookie, "Application.Chart.ChartTitle.Text",
                          "Pronto Xi Chart")
    message "put-property error for chart title"
  endif
  
  if not ole-put-property(cookie, "Application.Chart.HasTitle", TRUE)
    message "put-property error for has chart title text"
  endif
  if not ole-put-property(cookie, "Application.Chart.ChartTitle.Text",
                          "Pronto Xi Chart")
    message "put-property error for chart title"
  endif
  
  if not ole-put-property(cookie, "Application.Caption", "Pronto Xi Chart")
    message "put-property error for window caption"  ole-status()
            ole-error-description(ole-status())

  if not ole-get-property(cookie, "Axes", axes-id,1)

  if not ole-put-property(axes-id, "HasTitle", TRUE)

  if not ole-release(axes-id)
    endif
  endif
  
  
end-procedure //fill-sheet-font

procedure fill-a-cell
parameters
  p-row-col pic x(20)

Populating an Microsoft Excel Spreadsheet and Creating a Chart

This code example populates an Microsoft Excel spreadsheet with predefined data and creates a chart from the data in the same spreadsheet:

//////////////////////////////////////////////////////
// Sample 4GL program for OLE automation - Excel scatter graph
///////////////////////////////////////////////////////
        
#define APPNAME "Excel.Application"
#define CLSCTX_LOCAL_SERVER 4
        
field inst-id   type ole-interface
field wb-id     type ole-interface
field ws-id     type ole-interface
field chart-id  type ole-interface
field ver       pic x(60)
field dummy-ret pic 999

procedure main
  set inst-id = ole-get-active-object(APPNAME)
  if inst-id <= 0
    set inst-id = ole-create-instance(APPNAME, CLSCTX_LOCAL_SERVER)
    if inst-id <= 0
      abort "Cannot create instance - error ", ole-status()
    end
  end
  if not ole-get-property(inst-id, "Application.Version", ver)
    abort "Could not get version property: ", ole-status(), " - ",
            ole-error-description(ole-status())
  end
  if not ole-call-method(inst-id, "Application.Workbooks.Add", dummy-ret)
    abort "call-method workbooks.add failed"
  end
  if not ole-call-method(inst-id, "Application.Charts.Add", dummy-ret)
    abort "call-method charts.add failed"
  end
  // Get first workbook
  if not ole-get-property(inst-id, "Application.Workbooks.Item", wb-id, 1)
    abort "Could not get workbook id - error ", ole-status()
  end
  // Get first worksheet in workbook
  if not ole-get-property(wb-id, "WorkSheets.Item", ws-id, 1)
    abort "Could not get worksheets item id - error ", ole-status()
  end
  if not ole-get-property(wb-id, "Charts.Item", chart-id, 1)
    abort "Could not get charts item id - error ", ole-status()
  end
  do fill-sheet
  do chart-sheet
  if not ole-put-property(inst-id, "Application.Visible", TRUE)
    abort "Could not put-property Visible"
  end
  if ole-release(chart-id)
  endif
  if ole-release(ws-id)
  endif
  if ole-release(wb-id)
  endif
  if ole-release(inst-id)
  endif
  end-procedure //main ----------------------------------
        
procedure chart-sheet
  local field range-id type ole-interface
  if not ole-put-property(chart-id, "ChartType", 74)
    abort "put-property error for ChartType"
  endif
  if not ole-get-property(ws-id, "Range", range-id, "A1", "B10")
    abort "Could not get range id - error ", ole-status()
  end
  message "range is " range-id
  if not ole-call-method(chart-id, "SetSourceData", dummy-ret, range-id, 2)
    message "call-method SetSourceData failed"
            ole-error-description(ole-status())
  end
  if not ole-put-property(chart-id, "HasTitle", TRUE)
    message "put-property error for HasTitle"
            ole-error-description(ole-status())
  endif
  if not ole-put-property(chart-id, "ChartTitle.Text",
                          "Sample XY Scatter Chart")
    message "put-property error for Title"
  endif
  if not ole-put-property(chart-id, "HasLegend", TRUE)
    message "put-property error for HasLegend"
  endif

Replacing Bookmarks and Inserting Text into Word Documents

This code example opens a new Word document (.doc) or Word template (.dot) and provides samples of how to replace bookmarks and insert text. This code example also contains samples of other Word methods and properties.

/////////////////////////////////////////////////
// Sample 4GL program for OLE automation - Word. //
// To find the many Word ole properties and functions, go to a Word // Document and follow this menu path.
//
// Tools > Macros > VB editor > Help > MS VB help > MS Word VB Reference
// //
/////////////////////////////////////////////////

#define APPNAME "Word.Application"
#include "../../../include/i8word.spl"

field inst-id      type ole-interface
field wb-id        type ole-interface
field ws-id        type ole-interface
field ver          pic x(60)
field dummy-ret    pic 999
field dummy-ret-a  pic x(100)
      ws-range-id  pic s9(8)
      ws-xl        pic s9(8)
      i            pic 999
      tmp-bookmark type string
      tmp-index    type numeric
procedure main

  // Open up Word session
  set inst-id = ole-get-active-object(APPNAME)
  if inst-id <= 0
    set inst-id = ole-create-instance(APPNAME, CLSCTX_LOCAL_SERVER)
    if inst-id <= 0
      abort "Cannot create instance - error ", ole-status()
    end
  end
  if not ole-get-property(inst-id, "Application.Version", ver)
    abort "Could not get version property: ", ole-status()
  end

  // Create a new document from the template doc
  if not ole-call-method(inst-id, "Application.Documents.Add",
                                   dummy-ret ,"C:\sms.doc")
    message "Template document not added"

    // Just open up a blank document
    if not ole-call-method(inst-id, "Application.Documents.Add", dummy-ret)
      message "no add"
    end
  end

  // Display the document
  set ws-xl = ole-put-property(inst-id, "Application.Visible", TRUE)

  // This block is a way of replacing bookmarks if they exist. 
  if ole-get-property(inst-id, "ActiveDocument.Bookmarks",wb-id)
    if ole-get-property(wb-id,"Count",i)
      for tmp-index = i down to 1
        if ole-call-method(wb-id, "Item",ws-range-id,tmp-index)
          if ole-get-property(ws-range-id, "Name",tmp-bookmark)
            // You could compare tmp-bookmark to the actual bookmark // before subbing text. 
            if ole-call-method(ws-range-id,"Select",dummy-ret)
              set ws-xl = ole-call-method(inst-id,"Selection.TypeText",
                                          dummy-ret,"Bookmark replacing text")
            endif
              endif
            endif
         end-for
     endif
  endif

  // Move cursor down 10 lines
  set ws-xl = ole-call-method(inst-id,"Selection.MoveDown",dummy-ret,wdLine,10)

  // Insert text
  set ws-xl = ole-call-method(inst-id,"Selection.TypeText",dummy-ret,
                              "Inserted text at cursor position")

  // Set the printer(uses default if not set)
  if ole-put-property(inst-id, "ActivePrinter","lp0")
  endif

  // Print spreadsheet // IF OLE-CALL-METHOD(inst-id,"Application.PrintOut",dummy-ret) // ENDIF // Help in here will show you all the commands required to manipulate Word //(select help from the visual basic editor)
  if ole-put-property(inst-id,"ShowVisualBasicEditor",TRUE)
  endif

  //Close spreadsheet(no save)
  message "After this the word doco will close"
  if ole-call-method(inst-id, "Application.Quit", dummy-ret,wdDoNotSaveChanges)
  endif

  // Release all OLE interfaces
  if ole-release(ws-id)
  endif
  if ole-release(wb-id)
  endif
  if ole-release(inst-id)
  endif
  end-procedure //main ---------------------
Embedding Web Components into a Pronto Xi Screen

This code example uses web components inside a Pronto Xi screen; for example, using Microsoft Windows components to simulate Microsoft Excel to allow entry of data for a graph and display the details in graph format:

////////////////////////////////////////
//
//
// Sample 4GL program for OLE automation - Excel scatter graph
//
//
///////////////////////////////////////
        
field sheet-id  type ole-interface
      graph-id  type ole-interface
      chart-id  type ole-interface
      ver       pic x(60)
      dummy-ret pic 999

procedure main
  do main-control
  if ole-release(sheet-id)
  end
  if ole-release(graph-id)
  end
end-procedure //main ----------------------
        
procedure main-control
window
  no-hide
  set sheet-id = ole-create-control("OWC10.Spreadsheet.10", 14, 1, 8, 80)
  if sheet-id <= 0
    abort "Cannot create spreadsheet control - error ", 
            ole-error-description(ole-status())
    end
    set graph-id = ole-create-control("OWC10.Chartspace.10", 2, 1, 10, 80)
    if graph-id <= 0
      abort "Cannot create chartspace control - error ", 
              ole-error-description(ole-status())
    end
    do fill-sheet
    do fill-graph
    pause
end-procedure //main-control ------------------
        
procedure fill-sheet
  if not ole-call-method(sheet-id, "ActiveSheet.Cells.Clear", dummy-ret)
    abort "call-method cells clear failed - ",
            ole-error-description(ole-status())
  end

  do fill-sheet-cell parameters 2, 1, "Consignment"
  do fill-sheet-cell parameters 3, 1, "On Hand"
  do fill-sheet-cell parameters 4, 1, "Internal"
  do fill-sheet-cell parameters 5, 1, "Current"
  do fill-sheet-cell parameters 6, 1, "Back"
  do fill-sheet-cell parameters 7, 1, "Forward"
  do fill-sheet-cell parameters 8, 1, "Transit"
  do fill-sheet-cell parameters 9, 1, "Available"
  do fill-sheet-cell parameters 1, 2, "Balances"
  do fill-sheet-cell parameters 2, 2, "0"
  do fill-sheet-cell parameters 3, 2, "6"
  do fill-sheet-cell parameters 4, 2, "0"
  do fill-sheet-cell parameters 5, 2, "2"
  do fill-sheet-cell parameters 6, 2, "0"
  do fill-sheet-cell parameters 7, 2, "0"
  do fill-sheet-cell parameters 8, 2, "0"
  do fill-sheet-cell parameters 9, 2, "4"
end-procedure //fill-sheet ---------------------
        
procedure fill-sheet-cell
  parameters are y   pic 99
                   x   pic 99
                   txt pic x(40)
  local field cells-id type ole-interface
  if not ole-get-property(sheet-id, "ActiveSheet.Cells", cells-id, y, x)
    abort "Could not get cells id - error ", ole-status()
  end
  if not ole-put-property(cells-id, "Value", txt)
    abort "put property value failed "
            ole-error-description(ole-status())
  end
  if ole-release(cells-id)
  endif
end-procedure //fill-sheet-cell ------------------
        
procedure fill-graph
  if not ole-call-method(graph-id, "Clear", dummy-ret)
    abort "call-method sheet clear failed - "
            ole-error-description(ole-status())
  end
  if not ole-call-method(graph-id, "Charts.Add", dummy-ret)
    abort "call-method charts.add failed - "
            ole-error-description(ole-status())
  end
  // connect the graph to the spreadsheet data source
#if 1
  if not ole-put-property(graph-id, "DataSource", sheet-id)
    abort "Could not put-property Datasource"
  end

#endif
  // Get ID of first chart
  if not ole-get-property(graph-id, "Charts.Item", chart-id, 0)
    abort "Could not get charts.item id - ",
            ole-error-description(ole-status())
  end
  // fill interior with grey
  if not ole-put-property(chart-id, "Interior.Color", 12632256)
    abort "Could not put-property Interior.Color"
  end
  do set-chart-series-collection
  if not ole-put-property(chart-id, "HasLegend", 0)
    abort "Could not put-property HasLegend"
  end
  if not ole-put-property(chart-id, "HasTitle", 1)
    abort "Could not put-property HasTitle"
  end
  do set-chart-axes parameter is -3
  if not ole-put-property(chart-id, "Title.Caption", "Melbourne Balances")
    abort "Could not put-property Title.Caption"
  end
  if ole-release(chart-id)
  end
end-procedure //fill-graph -------------------------
        
procedure set-chart-axes
parameters
  axis-type pic s9(10)
local field axis-id pic s9(8)
  if not ole-get-property(chart-id, "Axes", axis-id, axis-type)
    abort "Could not get axis id - error ", 
            ole-error-description(ole-status())
  end
  if not ole-put-property(axis-id, "NumberFormat", "0")
    abort "put-property error for numberformat -"
            ole-error-description(ole-status())
  endif
  if not ole-put-property(axis-id, "MajorUnit", 1)
    abort "put-property error for majorunit"
  endif
end-procedure //set-chart-axes ----------------------
        
procedure set-chart-series-collection
local field series-id type ole-interface
  
  if not ole-call-method(chart-id, "SeriesCollection.Add", dummy-ret)
    abort "Could not call-method SeriesCollection.Add ", ole-status()
  end
  if not ole-get-property(chart-id, "SeriesCollection.Item", series-id, 0)
    abort "Could not get series id - error ", ole-status()
  end
  // series name
  if not ole-call-method(series-id, "SetData", dummy-ret, 0, 0, "B1")
    abort "Could not call-method SetData 1", ole-status()
  end
  // categories
  if not ole-call-method(series-id, "SetData", dummy-ret, 1, 0, "A2:A9")
    abort "Could not call-method SetData 2", ole-status()
  end
  // values
  if not ole-call-method(series-id, "SetData", dummy-ret, 2, 0, "B2:B9")
    abort "Could not call-method SetData 3", ole-status()
  end
  if not ole-release(series-id)
  endif
 end-procedure //set-chart-series-collection --------------
